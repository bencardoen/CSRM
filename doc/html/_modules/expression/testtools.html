[docs]    def testGenerator(self):
        l = [1,2,3, 4]
        for k in randomizedConsume(l, seed=0):
            assert(k not in l)
        assert(len(l) == 0)

        l = [1,2,3,4]
        lold = l[:]
        for k in permutate(l, seed=0):
            assert(k in l)
        assert(l != lold)



[docs]    def testFlatten(self):
        a = [1,2, [1,2], [2,3,[5]]]
        b = flatten(a)

            self.assertFalse(isinstance(x, list))
        self.assertEqual(b, [1,2,1,2,2,3,5])
